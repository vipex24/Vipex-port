-- Combat System | Server-authoritative with client prediction local CombatSystem = {} CombatSystem.__index = CombatSystem function CombatSystem.new(config) local self = setmetatable({}, CombatSystem) self.hitboxSize = config.hitboxSize or Vector3.new(5, 5, 5) self.cooldowns = {} self.combos = {} self.damageMultipliers = config.multipliers or {} return self end function CombatSystem:ProcessAttack(attacker, attackType) if self:IsOnCooldown(attacker) then return false end local combo = self:GetComboState(attacker) local damage = self:CalculateDamage(attackType, combo) local targets = self:CastHitbox(attacker) for _, target in targets do self:ApplyDamage(target, damage) self:ApplyKnockback(target, attacker, combo.force) self:PlayHitEffects(target.Position) end self:AdvanceCombo(attacker) self:StartCooldown(attacker, combo.cooldown) return true end
-- Inventory System | Replicated, slot-based with stacking local Inventory = {} Inventory.__index = Inventory function Inventory.new(player, maxSlots) local self = setmetatable({}, Inventory) self.player = player self.maxSlots = maxSlots or 36 self.slots = {} self.onChange = Signal.new() return self end function Inventory:AddItem(itemId, quantity) local itemData = ItemDatabase:GetItem(itemId) if not itemData then return false, "Invalid item" end if itemData.stackable then local existing = self:FindStack(itemId) if existing then existing.quantity += quantity self.onChange:Fire("update", existing) return true end end local slot = self:FindEmptySlot() if not slot then return false, "Inventory full" end self.slots[slot] = { id = itemId, quantity = quantity } self.onChange:Fire("add", self.slots[slot]) return true end
-- DataStore Wrapper | Session locking, auto-save, migration local DataManager = {} local DataStoreService = game:GetService("DataStoreService") local AUTOSAVE_INTERVAL = 120 local DATA_VERSION = 3 local DEFAULT_DATA = { version = DATA_VERSION, coins = 0, level = 1, exp = 0, inventory = {}, settings = { music = true, sfx = true }, stats = { playtime = 0, kills = 0 }, } function DataManager:LoadPlayerData(player) local key = "Player_" .. player.UserId local success, data = pcall(function() return self.store:GetAsync(key) end) if success and data then data = self:MigrateData(data) data = self:Reconcile(data, DEFAULT_DATA) else data = DeepCopy(DEFAULT_DATA) end self.cache[player] = data self:StartAutoSave(player) return data end
-- UI Component Framework | Reactive, animated, reusable local UIComponent = {} UIComponent.__index = UIComponent function UIComponent.new(config) local self = setmetatable({}, UIComponent) self.frame = Instance.new("Frame") self.state = Observable.new(config.initialState or {}) self.children = {} self.tweens = {} return self end function UIComponent:Animate(property, target, duration) local tweenInfo = TweenInfo.new( duration or 0.3, Enum.EasingStyle.Quart, Enum.EasingDirection.Out ) local tween = TweenService:Create( self.frame, tweenInfo, { [property] = target } ) tween:Play() return tween end function UIComponent:BindState(key, callback) self.state:Observe(key, function(newVal, oldVal) callback(self.frame, newVal, oldVal) end) end function UIComponent:Show() self.frame.Visible = true self:Animate("GroupTransparency", 0, 0.25) self:Animate("Position", self.targetPos, 0.35) end
